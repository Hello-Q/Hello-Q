  • Vue.js 安装

    • CDN

        <!--对于制作原型或学习，你可以这样使用最新版本-->
  <script src="https://cdn.jsdelivr.net/npm/vue"></script>
  <!--对于生产环境，我们推荐链接到一个明确的版本号和构建文件，以避免新版本造成的不可预期的破坏：-->
  <script src="https://cdn.jsdelivr.net/npm/vue@2.6.10/dist/vue.js"></script>

  • 实例化Vue对象

    • js文件

        // 实例化Vue对象
  new Vue({
  el: '#vue-app',
  data:{
      name: '米斯特五'
  }
  })
  /* 
  el: element 需要获取的元素,一定是html中的根容器
  data: 用于数据的存储
  */

    • html文件

        <!DOCTYPE html>
  <html>
      <head>
          <meta charset="utf-8">
          <title></title>
          <script src="https://cdn.jsdelivr.net/npm/vue"></script>
      </head>
      <body>
          <div id="vue-app">
              <h1>{{name}}</h1>
          </div>
          <script src="app.js" type="text/javascript" charset="utf-8"></script>
      </body>
  </html>

  • 数据和方法

    • 数据

      • js文件中的数据

        在data中声明

          new Vue({
  el: '#vue-app',
  // 数据放在data中
  data:{
      name: '米斯特五',
      job: 'web开发'
      },
  })

      • 在html中使用

        包裹在双花括号中变量名字

          <div id="vue-app">
      <h1>Name: {{name}}</h1>
      <h1>Job: {{job}}</h1>
  </div>

    • 方法

      放在method

        new Vue({
      el: '#vue-app',
      methods:{
          greet: function(){
              return 'Good Morning' ;
          }
      }
  })

      使用

        <div id="vue-app">
      <h1>{{ greet() }}</h1>
  </div>

  • 属性绑定

    使用v-bind 进行属性绑定

      <div id="vue-app">
      <a v-bind:href="website">baidu</a>
      <input type="text" v-bind:value="name">
      <p v-html="websiteTag"></p>
  </div>

  • 事件

    使用v-on:xxx=”method” 绑定事件
    也可使用@代替v-on:

      <div id="vue-app">
      <h1>Events</h1>
      <button type="button" v-on:click="add"> 长一岁</button>
      <button type="button" v-on:click="subtract"> 减一岁</button>
      <p>My age is {{age}}</p>
  </div>

  • 修饰符

    .stop
    .prevent
    .capture
    .self
    .once
    .passive

    例子

      <!-- 阻止单击事件继续传播 -->
  <a v-on:click.stop="doThis"></a>

  <!-- 提交事件不再重载页面 -->
  <form v-on:submit.prevent="onSubmit"></form>

  <!-- 修饰符可以串联 -->
  <a v-on:click.stop.prevent="doThat"></a>

  <!-- 只有修饰符 -->
  <form v-on:submit.prevent></form>

  <!-- 添加事件监听器时使用事件捕获模式 -->
  <!-- 即元素自身触发的事件先在此处理，然后才交由内部元素进行处理 -->
  <div v-on:click.capture="doThis">...</div>

  <!-- 只当在 event.target 是当前元素自身时触发处理函数 -->
  <!-- 即事件不是从内部元素触发的 -->
  <div v-on:click.self="doThat">...</div>
