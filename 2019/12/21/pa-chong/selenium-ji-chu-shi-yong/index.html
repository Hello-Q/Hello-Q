一个简单的程序

# coding=utf-8

from selenium import webdriver
import time

# 使用谷歌驱动
driver = webdriver.Chrome()
url = "https://www.baidu.com/"
# 发起请求
driver.get(url)
time.sleep(1.5)
# 在搜索框中输入“python”
driver.find_element_by_id("kw").send_keys("python")
time.sleep(2)
# 点击“百度一下”
driver.find_element_by_xpath("//input[@type='submit']").click()
time.sleep(4)
driver.close()
